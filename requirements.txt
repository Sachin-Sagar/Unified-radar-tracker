# This file lists the Python packages required to run the Unified Radar Tracker.
# You can install all of them at once by running:
# pip install -r requirements.txt

numpy           # Used for numerical operations, especially array manipulation for the point cloud and tracking data.
pyserial        # Required for serial communication with the radar hardware.
PyQt5           # The GUI toolkit used to create the live visualizer window and its components.
pyqtgraph       # Used for creating the real-time plots within the PyQt5 GUI.
scipy           # Used for scientific and technical computing, particularly for signal processing in this project.
